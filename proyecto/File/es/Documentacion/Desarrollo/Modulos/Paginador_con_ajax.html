Ejemplo de metodo paginador

   /** 
    * Presentar los movimientos abiertos
    */

   function movimientos_abiertos($id_proyecto=NULL) {
      global $gcm;
      $condicion_proyecto = ( $id_proyecto ) ? " AND p.id=".$id_proyecto : '';
      $campo_proyecto =  ( ! $id_proyecto ) ? " p.nombre ,' / '," : '';

      $sql = "SELECT CONCAT(".$campo_proyecto." t.nombre , ' / ', m.nombre) as Movimiento 
              FROM movimientos m, tiempos ti, proyectos p, tareas t
              WHERE t.idProyecto=p.id AND t.id=m.idTarea AND m.id=ti.idMovimiento ".$condicion_proyecto."
              AND (m.fechaFinal = null || m.fechaFinal = '0000-00-00 00:00:00')
              GROUP BY m.id
              ORDER BY ti.fechaFinal desc
              ";

      require_once(GCM_DIR.'lib/int/GcmPDO/lib/paginarPDO.php');

      $pd = new PaginarPDO($this->pdo, $sql, 'ma_');

      if ( $pd->validar() ) {
         $pd->generar_pagina('&m=trabajos&a=movimientos_abiertos&formato=ajax');
         }
      }
