00116    * Retrieve the coordinates of the given event relative to the center
00117    * of the widget.
00118    */
00119   fb.widgetCoords = function (event) {
00120     var x, y;
00121     var el = event.target || event.srcElement;
00122     var reference = fb.wheel;
00123 
00124     if (typeof event.offsetX != 'undefined') {
00125       // Use offset coordinates and find common offsetParent
00126       var pos = { x: event.offsetX, y: event.offsetY };
00127 
00128       // Send the coordinates upwards through the offsetParent chain.
00129       var e = el;
00130       while (e) {
00131         e.mouseX = pos.x;
00132         e.mouseY = pos.y;
00133         pos.x += e.offsetLeft;
00134         pos.y += e.offsetTop;
00135         e = e.offsetParent;
00136       }
00137 
00138       // Look for the coordinates starting from the wheel widget.
00139       var e = reference;
00140       var offset = { x: 0, y: 0 }
00141       while (e) {
00142         if (typeof e.mouseX != 'undefined') {
00143           x = e.mouseX - offset.x;
00144           y = e.mouseY - offset.y;
00145           break;
00146         }
00147         offset.x += e.offsetLeft;
00148         offset.y += e.offsetTop;
00149         e = e.offsetParent;
00150       }
00151 
00152       // Reset stored coordinates
00153       e = el;
00154       while (e) {
00155         e.mouseX = undefined;
00156         e.mouseY = undefined;
00157         e = e.offsetParent;
00158       }
00159     }
00160     else {
00161       // Use absolute coordinates
00162       var pos = fb.absolutePosition(reference);
00163       x = (event.pageX || 0*(event.clientX + $('html').get(0).scrollLeft)) - pos.x;
00164       y = (event.pageY || 0*(event.clientY + $('html').get(0).scrollTop)) - pos.y;
00165     }
00166     // Subtract distance to middle
00167     return { x: x - fb.width / 2, y: y - fb.width / 2 };
00168   }

00306   fb.hueToRGB = function (m1, m2, h) {
00307     h = (h < 0) ? h + 1 : ((h > 1) ? h - 1 : h);
00308     if (h * 6 < 1) return m1 + (m2 - m1) * h * 6;
00309     if (h * 2 < 1) return m2;
00310     if (h * 3 < 2) return m1 + (m2 - m1) * (0.66666 - h) * 6;
00311     return m1;
00312   }

00314   fb.RGBToHSL = function (rgb) {
00315     var min, max, delta, h, s, l;
00316     var r = rgb[0], g = rgb[1], b = rgb[2];
00317     min = Math.min(r, Math.min(g, b));
00318     max = Math.max(r, Math.max(g, b));
00319     delta = max - min;
00320     l = (min + max) / 2;
00321     s = 0;
00322     if (l > 0 && l < 1) {
00323       s = delta / (l < 0.5 ? (2 * l) : (2 - 2 * l));
00324     }
00325     h = 0;
00326     if (delta > 0) {
00327       if (max == r && max != g) h += (g - b) / delta;
00328       if (max == g && max != b) h += (2 + (b - r) / delta);
00329       if (max == b && max != r) h += (4 + (r - g) / delta);
00330       h /= 6;
00331     }
00332     return [h, s, l];
00333   }
